  import json

  def main(status_code: int, body, doc_name: str) -> dict:
      # 1) HTTP 失败直接判不存在（也可按需抛错）
      if int(status_code) != 200:
          return {"ok": False, "exists": False, "ready": False, "doc_id": "", "reason": f"http_{status_code}"}

      # 2) body 可能是 JSON 字符串，也可能已是对象
      payload = body
      if isinstance(body, str):
          s = body.strip()
          payload = json.loads(s) if s else {}

      docs = payload.get("data") or []
      total = int(payload.get("total") or 0)

      # 3) 严谨：做“精确命名匹配”，避免 keyword 模糊命中误判
      match = None
      for item in docs:
          if item.get("name") == doc_name:
              match = item
              break
          # 兼容：也有人用 doc_metadata 里 document_name
          for md in item.get("doc_metadata") or []:
              if md.get("name") == "document_name" and md.get("value") == doc_name:
                  match = item
                  break
          if match:
              break

      if not match:
          return {"ok": True, "exists": False, "ready": False, "total": total, "doc_id": ""}

      enabled = bool(match.get("enabled", True))
      archived = bool(match.get("archived", False))
      indexing_status = match.get("indexing_status") or ""
      display_status = match.get("display_status") or ""

      exists = enabled and not archived
      # “能检索”通常要求索引完成且可用
      ready = exists and indexing_status == "completed" and display_status == "available"

      return {
          "ok": True,
          "exists": exists,
          "ready": ready,
          "total": total,
          "doc_id": match.get("id") or "",
          "indexing_status": indexing_status,
          "display_status": display_status,
      }

  - 解一次 json.loads 后得到的仍然是字符串（里面才是真正的 JSON）

  - 自动处理 多层 JSON 字符串（最多尝试解 3 次）
  - 如果你不小心把整个 HTTP 输出对象传进来（包含 status_code/body/headers），也会自动取里面的 body
  - 如果解析失败，不会再抛异常，而是返回 ok=false + reason，方便你在 If-Else 里分支处理

  import json
  from typing import Any, Dict

  def _parse_json_maybe(value: Any) -> Dict[str, Any]:
      # 兼容：如果你把整个 HTTP 输出对象传进来了（里面还有 body 字段）
      if isinstance(value, dict) and "data" not in value and "body" in value:
          value = value.get("body")

      # 兼容：body 可能是“JSON 字符串套 JSON 字符串”，循环解几次
      for _ in range(3):
          if isinstance(value, str):
              s = value.strip()
              if not s:
                  return {}
              try:
                  value = json.loads(s)
                  continue
              except Exception:
                  # 不是 JSON 字符串就停止
                  break
          break

      return value if isinstance(value, dict) else {}

  def main(status_code: int, body, doc_name: str) -> dict:
      if int(status_code) != 200:
          return {"ok": False, "exists": False, "ready": False, "doc_id": "", "reason": f"http_{status_code}"}

      payload = _parse_json_maybe(body)
      if not payload:
          return {
              "ok": False,
              "exists": False,
              "ready": False,
              "doc_id": "",
              "reason": "body_not_json_object",
          }

      data_field = payload.get("data") or []
      if isinstance(data_field, list):
          docs = data_field
      elif isinstance(data_field, dict):
          docs = list(data_field.values())
      else:
          docs = []

      total = payload.get("total")
      try:
          total = int(total) if total is not None else len(docs)
      except Exception:
          total = len(docs)

      # 严谨：精确匹配 name / doc_metadata.document_name
      match = None
      for item in docs:
          if not isinstance(item, dict):
              continue

          if item.get("name") == doc_name:
              match = item
              break

          for md in item.get("doc_metadata") or []:
              if isinstance(md, dict) and md.get("name") == "document_name" and md.get("value") == doc_name:
                  match = item
                  break

          if match:
              break

      if not match:
          return {"ok": True, "exists": False, "ready": False, "total": total, "doc_id": ""}

      enabled = bool(match.get("enabled", True))
      archived = bool(match.get("archived", False))
      indexing_status = match.get("indexing_status") or ""
      display_status = match.get("display_status") or ""

      exists = enabled and not archived
      ready = exists and indexing_status == "completed" and display_status == "available"

      return {
          "ok": True,
          "exists": exists,
          "ready": ready,
          "total": total,
          "doc_id": match.get("id") or "",
          "indexing_status": indexing_status,
          "display_status": display_status,
      }
