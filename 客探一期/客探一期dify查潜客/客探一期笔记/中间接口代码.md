• 下面是一份按你给的固定 payload（chat-messages、不需要 query、只传 employee_code）写的 Flask 中间接口：前端先 POST 拿到 redirect_url，再把 iframe.src 指过去  
  即可（每次调用都会新建会话）。  
  app.py  
  
  import os  
  import re  
  import logging  
  from urllib.parse import urlparse, parse_qsl, urlencode, urlunparse  
  
  import requests  
  from flask import Flask, request, jsonify  
  
  app = Flask(__name__)  
  
  logging.basicConfig(level=os.getenv("LOG_LEVEL", "INFO"))  
  logger = logging.getLogger("dify-middle")  
  
  # ====== Config (ENV) ======  
  DIFY_API_URL = os.getenv("DIFY_API_URL", "http://dify.amer.cn/v1/chat-messages")  
  DIFY_CHAT_URL = os.getenv("DIFY_CHAT_URL", "http://dify.amer.cn/chat/qQrRtKvsKyEmBoho")  
  DIFY_API_KEY = os.getenv("DIFY_API_KEY", "")  # 必填：不要写死在代码里  
  
  DIFY_RESPONSE_MODE = os.getenv("DIFY_RESPONSE_MODE", "blocking")  
  TIMEOUT_CONNECT = float(os.getenv("DIFY_TIMEOUT_CONNECT", "3"))  
  TIMEOUT_READ = float(os.getenv("DIFY_TIMEOUT_READ", "20"))  
  
  # 可选：给接口加个门票，避免公网被刷爆（推荐上线时一定要配）  
  MIDDLE_TOKEN = os.getenv("MIDDLE_TOKEN", "")  # 若设置则必须携带  
  
  # CORS：建议配置成你们 CRM 的 Origin；临时也可以用 "*"  CORS_ALLOWED_ORIGINS = [o.strip() for o in os.getenv("CORS_ALLOWED_ORIGINS", "*").split(",") if o.strip()]  
  
  EMPLOYEE_CODE_RE = re.compile(r"^[0-9A-Za-z_-]{1,64}$")  
  
  
  def append_query(url: str, **params):  
      parts = urlparse(url)  
      query = dict(parse_qsl(parts.query, keep_blank_values=True))  
      for k, v in params.items():  
          if v is None:  
              continue  
          query[k] = str(v)  
      return urlunparse(parts._replace(query=urlencode(query, doseq=True)))  
  
  
  def add_cors_headers(resp):  
      origin = request.headers.get("Origin")  
      if not origin:  
          return resp  
  
      if CORS_ALLOWED_ORIGINS == ["*"]:  
          resp.headers["Access-Control-Allow-Origin"] = "*"  
      elif origin in CORS_ALLOWED_ORIGINS:  
          resp.headers["Access-Control-Allow-Origin"] = origin  
          resp.headers["Vary"] = "Origin"  
  
      resp.headers["Access-Control-Allow-Methods"] = "POST, OPTIONS"  
      resp.headers["Access-Control-Allow-Headers"] = "Content-Type, X-Middle-Token"  
      resp.headers["Access-Control-Max-Age"] = "600"  
      return resp  
  
  
  @app.after_request  
  def after_request(resp):  
      resp.headers.setdefault("Cache-Control", "no-store")  
      return add_cors_headers(resp)  
  
  
  @app.route("/healthz", methods=["GET"])  
  def healthz():  
      return "ok", 200  
  
  
  @app.route("/api/dify/session", methods=["POST", "OPTIONS"])  
  def create_dify_session():  
      # CORS 预检  
      if request.method == "OPTIONS":  
          return ("", 204)  
  
      # 可选鉴权：避免被扫/被刷  
      if MIDDLE_TOKEN:  
          token = request.headers.get("X-Middle-Token") or request.args.get("token")  
          if token != MIDDLE_TOKEN:  
              return jsonify({"error": "unauthorized"}), 401  
  
      if not DIFY_API_KEY:  
          logger.error("DIFY_API_KEY not set")  
          return jsonify({"error": "server_misconfigured"}), 500  
  
      body = request.get_json(silent=True) or {}  
      employee_code = str(body.get("employee_code") or "").strip()  
  
      if not employee_code:  
          return jsonify({"error": "missing employee_code"}), 400  
      if not EMPLOYEE_CODE_RE.match(employee_code):  
          return jsonify({"error": "invalid employee_code"}), 400  
  
      payload = {  
          "inputs": {"employee_code": employee_code},  
          "response_mode": DIFY_RESPONSE_MODE,  
          "user": f"sales_{employee_code}",  
      }  
  
      headers = {  
          "Authorization": f"Bearer {DIFY_API_KEY}",  
          "Content-Type": "application/json",  
      }  
  
      try:  
          resp = requests.post(  
              DIFY_API_URL,  
              headers=headers,  
              json=payload,  
              timeout=(TIMEOUT_CONNECT, TIMEOUT_READ),  
          )  
          resp.raise_for_status()  
          data = resp.json()  
      except requests.Timeout:  
          return jsonify({"error": "dify_timeout"}), 504  
      except Exception:  
          logger.exception("Dify call failed")  
          return jsonify({"error": "dify_call_failed"}), 502  
  
      conversation_id = data.get("conversation_id") or (data.get("data") or {}).get("conversation_id")  
      redirect_url = append_query(DIFY_CHAT_URL, conversation_id=conversation_id) if conversation_id else DIFY_CHAT_URL  
  
      return jsonify(  
          {  
              "employee_code": employee_code,  
              "conversation_id": conversation_id,  
              "redirect_url": redirect_url,  
          }  
      )  
  
  
  if __name__ == "__main__":  
      port = int(os.getenv("PORT", "8000"))  
      app.run(host="0.0.0.0", port=port)  
  
  前端调用方式（拿到 URL 再跳转/设置 iframe）：  
  
  const r = await fetch("https://mid.xxx.com/api/dify/session", {  
    method: "POST",  
    headers: {  
      "Content-Type": "application/json",  
      "X-Middle-Token": "YOUR_TOKEN" // 如果你后端配置了 MIDDLE_TOKEN  
    },  
    body: JSON.stringify({ employee_code: "E00123" })  
  });  
  const data = await r.json();  
  iframe.src = data.redirect_url;  
  
  两点你一定要提前确认（否则嵌入企业微信会直接“白屏”）：  
  
  - 如果你们 CRM 工作台是 https，iframe 里加载 http://dify.amer.cn/... 可能会被“混合内容”拦截；最好把 Dify 和中间层都提供 https（或用反代给 Dify/中间层加  
    TLS）。  
  - 你之前贴过 DIFY_API_KEY：建议立刻在 Dify 吊销重置，再用环境变量 DIFY_API_KEY 配置到服务器上。
