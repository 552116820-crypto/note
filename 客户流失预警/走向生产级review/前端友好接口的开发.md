    headers: {
      "Content-Type": "application/json",

      ...(init?.headers ?? {}),
    },
  });

  if (!response.ok) {
    const message = await response.text();
    throw new Error(`${response.status} ${message}`);
  }

  return response.json() as Promise<T>;
}
